Hippie Tab Control System using JQuery

This is an extendable tab system that only requires you have an ul element on the page. It uses jQuery or Zepto and ties into event listeners for each thing it does.

Set Up

    <html>
<head>
    <title>HippieTabs Demo</title>
    <link href="lib/hippietabs.css" rel="stylesheet"/>
</head>
<body>
    <ul class="HippieTabs"></ul>

    <script src="lib/jquery-1.10.1.min.js"></script>
    <script src="lib/hippietabs.js"></script>
    <script>

        hippieTabs = $('.HippieTabs').hippieTabs();

        hippieTabs.createTab("Keylime Pie","keylime_pie", "Very delicious")
        hippieTabs.createTab("Carrots", "carrots", "Very nutritious")
        hippieTabs.createTab("Water", "water", "My favorite drink")

    </script>
</body>
</html>

    Manipulating Tabs

    createTab

    hippieTabs.createTab(<title>,<tabid>);

    hippieTabs.createTab(<title>,<tabid>,<data>);

    Examples

    hippieTabs.createTab("Browse Music", "musicbrowser");

    hippieTabs.createTab("Search Library", "query1", "genre=Rock;artist=Miike Snow");

    removeTabById

    hippieTabs.removeTabById(<string>);

    Example

    hippieTabs.removeTabById("query1");

    activateTabById

    hippieTabs.activateTabById(<string>);

    Example

    hippieTabs.activateTabById("musicbrowser");

    Events

    htabcreate

    hippieTabs.on( "htabcreate", function(event, htabid, htabdata, tabtitle) {
    //called upon creation of a tab
})

    htabclose

    hippieTabs.on( "htabclose", function(event, htabid, htabdata) {
    //called when the tab begins closing animation
})

    htabactivate

    hippieTabs.on( "htabctivate", function(event, htabid, htabdata) {
    //called when a tab is selected
})
